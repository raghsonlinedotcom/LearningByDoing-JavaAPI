java.lang.String

Immutable sequence of characters. Common operations include search, slice, transform.

Summary: Returns the number of char values in this string.

// String.length()
public class Demo {
  public static void main(String[] args) {
    String s = "Hello, Java!";
    System.out.println(s.length());
  }
}
13

Summary: Returns a new string that is a substring of this string from begin (inclusive) to end (exclusive).

public class Demo {
  public static void main(String[] args) {
    String s = "Transaction#INV-2025";
    System.out.println(s.substring(0, 12));  // "Transaction"
    System.out.println(s.substring(13));     // "INV-2025"
  }
}
Transaction INV-2025

java.lang.Math

Basic numeric operations and utilities.

public class Demo {
  public static void main(String[] args) {
    System.out.println(Math.max(10, 25));   // 25
    System.out.println(Math.min(10, 25));   // 10

  }
}
25 10 1024.0 12.0 3 true

java.lang.Objects & System

Null checks, safe equals, hashing; environment/system utilities.

import java.util.Objects;
import java.util.Arrays;

public class Demo {
  public static void main(String[] args) {
    String name = "Raghav";
    System.out.println(Objects.requireNonNull(name));
    System.out.println(Objects.equals("a", "a"));      // true
    System.out.println(Objects.hash("x", 1, true));    // stable hash (example)
    System.out.println(System.getProperty("os.name")); // e.g., "Mac OS X"
    String[] src = {"A","B","C","D"};
    String[] dst = new String[2];
    System.arraycopy(src, 1, dst, 0, 2);
    System.out.println(Arrays.toString(dst));          // [B, C]
  }
}
Raghav true <some hash> <your OS name> [B, C]

java.util.List (ArrayList)

Ordered, index-based collection; allows duplicates.

import java.util.*;

public class Demo {
  public static void main(String[] args) {
    List<String> list = new ArrayList<>();
    list.add("Apple");
    list.add("Orange");
    list.add("Banana");
    System.out.println(list.get(1));         // Orange
    System.out.println(list.contains("Banana")); // true
    list.sort(Comparator.naturalOrder());    // Java 8+
    System.out.println(list);                // [Apple, Banana, Orange]
  }
}
Orange true [Apple, Banana, Orange]

java.util.Map (HashMap)

Key-value mapping with fast average-time operations.

import java.util.*;

public class Demo {
  public static void main(String[] args) {
    Map<String, Integer> counts = new HashMap<>();
    counts.put("apple", 2);
    counts.put("orange", 1);
    System.out.println(counts.get("apple"));           // 2
    System.out.println(counts.containsKey("banana"));  // false
    counts.computeIfAbsent("banana", k -> 0);
    counts.merge("banana", 3, Integer::sum);           // 3
    counts.forEach((k,v) -> System.out.println(k + "=" + v));
  }
}
2 false apple=2 orange=1 banana=3

Collections, Arrays, Optional

Helpers for collections/arrays and an expressive container for optional values.

import java.util.*;

public class Demo {
  public static void main(String[] args) {
    List<Integer> nums = new ArrayList<>(List.of(3,1,4,1,5));
    Collections.sort(nums);
    System.out.println(nums);           // [1,1,3,4,5]
    Collections.reverse(nums);
    System.out.println(nums);           // [5,4,3,1,1]
    System.out.println(Collections.frequency(nums, 1)); // 2
    List<Integer> ro = Collections.unmodifiableList(nums);
    System.out.println(ro);
  }
}

java.time.LocalDate

import java.time.*;

public class Demo {
  public static void main(String[] args) {

LocalDateTime & Duration

Date-time without zone; machine-intervals between instants or temporal objects.

import java.time.*;
import java.time.format.DateTimeFormatter;

public class Demo {
  public static void main(String[] args) {
    LocalDateTime now = LocalDateTime.now();
    LocalDateTime later = now.plusHours(5).minusDays(1);
    Duration d = Duration.between(now, later);
    System.out.println(d.toHours()); // 5 - 24 if minusDays(1) first? Here computed on ordered times.
    System.out.println(now.format(DateTimeFormatter.ISO_LOCAL_DATE_TIME));
  }
}
<hours difference> <current ISO date-time>

java.util.stream.Stream

Functional-style operations on sequences: map, filter, reduce, collect.

import java.util.*;
import java.util.stream.*;

public class Demo {
  public static void main(String[] args) {
    List<String> words = List.of("zebra","apple","mango","apple","kiwi");
    String joined = words.stream()
      .filter(w -> w.length() >= 5)     // filter
      .map(String::toUpperCase)         // map
      .distinct()                       // distinct
      .sorted()                         // sorted
      .collect(Collectors.joining(", "));
    System.out.println(joined);         // APPLE, MANGO, ZEBRA

    int total = IntStream.rangeClosed(1, 5).reduce(0, Integer::sum);
    System.out.println(total);          // 15
  }
}
APPLE, MANGO, ZEBRA 15
